  175.     function get_objects_with_tag($tag, $offset = 0, $limit = 100, $tagger_id = NULL) {
    176. 

  176.         if(!isset($tag)) {
    177. 

  177.             return false;
    178. 

  178.         }

  180.         $tag = $db->qstr($tag, get_magic_quotes_gpc());
    181. 

  181.  
    182. 

  182.         if(isset($tagger_id) && ($tagger_id > 0)) {
    183. 

  183.             $tagger_sql = "AND tagger_id = $tagger_id";
    184. 

  184.         } else {
    185. 

  185.             $tagger_sql = "";
    186. 

  186.         }

  187.         $prefix = $this->_table_prefix;

  188.  
    189.

    190. 

  190.             FROM ${prefix}freetagged_objects INNER JOIN ${prefix}freetags ON (tag_id = id)
    191. 

  191.             WHERE tag = $tag
    192. 

  192.             $tagger_sql
    193. 

  193.             ORDER BY object_id ASC
    194. 

  194.             LIMIT $offset, $limit
    195. 

  195.             ";

    197. 

  197.         $retarr = array();

  198.         while(!$rs->EOF) {

  263.      function get_objects_with_tag_combo($tagArray, $offset = 0, $limit = 100, $tagger_id = NULL) {
    264. 

  264.         if (!isset($tagArray) || !is_array($tagArray)) {
    265. 

  265.             return false;
    266. 

  266.         }
    267. 

  267.         $db = &$this->db;
    268. 

  268.         $retarr = array();
    269. 

  269.         if (count($tagArray) == 0) {
    270. 

  270.             return $retarr;
    271. 

  271.         }
    272. 

  272.         if(isset($tagger_id) && ($tagger_id > 0)) {
    273. 

  273.             $tagger_sql = "AND tagger_id = $tagger_id";
    274. 

  274.         } else {
    275. 

  275.             $tagger_sql = "";
    276. 

  276.         }
    277. 

  277.  
    278. 

  278.         foreach ($tagArray as $key => $value) {
    279. 

  279.             $tagArray[$key] = $db->qstr($value, get_magic_quotes_gpc());
    280. 

  280.         }
    281. 

  281.  
    282. 

  282.         $tagArray = array_unique($tagArray);
    283. 

  283.         $tag_sql = join(",", $tagArray);
    284. 

  284.         $numTags = count($tagArray);
    285. 

  285.         $prefix = $this->_table_prefix;
    286. 

  286.  
    287. 

  287.         // We must adjust for duplicate normalized tags appearing multiple times in the join by 
    288. 

  288.         // counting only the distinct tags. It should also work for an individual user.
    289. 

  289.  
    290. 

  290.         $sql = "SELECT ${prefix}freetagged_objects.object_id, tag, COUNT(DISTINCT tag) AS uniques
    291. 

  291.             FROM ${prefix}freetagged_objects 
    292. 

  292.             INNER JOIN ${prefix}freetags ON (${prefix}freetagged_objects.tag_id = ${prefix}freetags.id)
    293. 

  293.             WHERE ${prefix}freetags.tag IN ($tag_sql)
    294. 

  294.             $tagger_sql
    295. 

  295.             GROUP BY ${prefix}freetagged_objects.object_id
    296. 

  296.             HAVING uniques = $numTags
    297. 

  297.             LIMIT $offset, $limit
    298. 

  298.             ";
    299. 

  299.         $this->debug_text("Tag combo: " . join("+", $tagArray) . " SQL: $sql");
    300. 

  300.         $rs = $db->Execute($sql) or die("Error: $sql");
    301. 

  301.         while(!$rs->EOF) {
    302. 

  302.             $retarr[] = $rs->fields['object_id'];
    303. 

  303.             $rs->MoveNext();
    304. 

  304.         }
    305. 

  305.         return $retarr;
    306.

    767.

    768.

    769. 

  769.  
    770.
